281  Optimizer::getInterpretedTemplates(Structure *structure)
282  {
283  // Stop any running jobs associated with this structure
284  m_opt->queueInterface()->stopJob(structure);
285 
286  // Lock
287  QReadLocker locker (structure->lock());
288 
289  // Check optstep info
290  int optStep = structure->getCurrentOptStep();
291 
292  Q_ASSERT_X(optStep <= m_opt->optimizer()->getNumberOfOptSteps(),
293  Q_FUNC_INFO, QString("OptStep of Structure %1 exceeds "
294  "number of known OptSteps (%2, limit %3).")
295  .arg(structure->getIDString()).arg(optStep)
296  .arg(m_opt->optimizer()->getNumberOfOptSteps()).toStdString().c_str());
297 
298  // Unlock for optimizer calls
299  locker.unlock();
300 
301  // Build hash
302  QHash<QString, QString> hash;
303  QStringList filenames = m_templates.keys();
304  QString contents;
305  for (QStringList::const_iterator
306  it = filenames.constBegin(),
307  it_end = filenames.constEnd();
308  it != it_end;
309  it++) {
310  hash.insert((*it), m_opt->interpretTemplate(m_templates.value(*it)
311  .at(optStep - 1),
312  structure));
313  }
314  // QueueInterface templates
315  filenames = m_QITemplates.keys();
316  for (QStringList::const_iterator
317  it = filenames.constBegin(),
318  it_end = filenames.constEnd();
319  it != it_end;
320  it++) {
321  hash.insert((*it), m_opt->interpretTemplate(m_QITemplates.value(*it)
322  .at(optStep - 1),
323  structure));
324  }
325 
326  return hash;
327  }

393  bool Optimizer::update(Structure *structure)
394  {
395  // lock structure
396  QWriteLocker locker (structure->lock());
397 
398  structure->stopOptTimer();
399 
400  // Copy remote files over, other prep work:
401  locker.unlock();
402  bool ok = m_opt->queueInterface()->prepareForStructureUpdate(structure);
403  locker.relock();
404  if (!ok) {
405  m_opt->warning(tr("Optimizer::update: Error while preparing to update structure %1")
406  .arg(structure->getIDString()));
407  return false;
408  }
409 
410  // Try to read all files in outputFileNames
411  ok = false;
412  for (int i = 0; i < m_outputFilenames.size(); i++) {
413  if (read(structure,
414  structure->fileName() + "/" + m_outputFilenames.at(i))) {
415  ok = true;
416  break;
417  }
418  }
419  if (!ok) {
420  m_opt->warning(tr("Optimizer::Update: Error loading structure at %1")
421  .arg(structure->fileName()));
422  return false;
423  }
424 
425  structure->setJobID(0);
426  locker.unlock();
427  return true;

451  bool Optimizer::read(Structure *structure,
452  const QString & filename) {
453  // Test filename
454  QFile file (filename);
455  if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
456  return false;
457  }
458  file.close();
459 
460  // Read in OBMol
461  //
462  // OpenBabel::OBConversion;:ReadFile calls a singleton error class
463  // that is not thread safe. Hence sOBMutex is necessary.
464  m_opt->sOBMutex->lock();
465  OBConversion conv;
466  OBFormat* inFormat = conv.FormatFromExt(QString(QFile::encodeName(filename.trimmed())).toAscii());
467 
468  if ( !inFormat || !conv.SetInFormat( inFormat ) ) {
469  m_opt->warning(tr("Optimizer::read: Error setting openbabel format for file %1")
470  .arg(filename));
471  m_opt->sOBMutex->unlock();
472  return false;
473  }
474 
475  OBMol obmol;
476  if (!conv.ReadFile( &obmol, QString(QFile::encodeName(filename)).toStdString())) {
477  m_opt->error(tr("Cannot update structure %1 from file \n%2")
478  .arg(structure->getIDString())
479  .arg(QString(QFile::encodeName(filename))));
480  m_opt->sOBMutex->unlock();
481  return false;
482  }
483  m_opt->sOBMutex->unlock();
484 
485  // Extract data from obmol and update structure
486  double energy=0.0;
487  double enthalpy=0.0;
488  QList<unsigned int> atomicNums;
489  QList<Eigen::Vector3d> coords;
490  Eigen::Matrix3d cellMat = Eigen::Matrix3d::Zero();
491 
492  // Ensure that there are the correct number of atoms in the
493  // structure
494  while (structure->numAtoms() < obmol.NumAtoms())
495  structure->addAtom();
496  while (structure->numAtoms() > obmol.NumAtoms())
497  structure->removeAtom(structure->atoms().last());
498 
499  // Atomic data
500  FOR_ATOMS_OF_MOL(atm, obmol) {
501  coords.append(Eigen::Vector3d(atm->x(), atm->y(), atm->z()));
502  atomicNums.append(atm->GetAtomicNum());
503  }
504 
505  // energy/enthalpy
506  if (obmol.HasData("Enthalpy (kcal/mol)")) {
507  enthalpy = QString(obmol.GetData("Enthalpy (kcal/mol)")->GetValue().c_str()
508  ).toDouble() * KCAL_PER_MOL_TO_EV;
509  }
510  // Convert energy to eV (Avogadro::Molecule expects kcal/mol, but
511  // the updateAnd*History functions below want eV)
512  energy = obmol.GetEnergy() * KCAL_PER_MOL_TO_EV;
513 
514  // Cell
515  OBUnitCell *cell = static_cast<OBUnitCell*>(obmol.GetData(OBGenericDataType::UnitCell));
516 
517  if (cell != NULL) {
518  for (int i = 0; i < 3; i++) {
519  for (int j = 0; j < 3; j++) {
520  cellMat(i,j) = cell->GetCellMatrix().Get(i,j);
521  }
522  }
523  }
524 
525  if (m_opt->isStarting) {
526  structure->updateAndSkipHistory(atomicNums, coords, energy, enthalpy, cellMat);
527  }
528  else {
529  structure->updateAndAddToHistory(atomicNums, coords, energy, enthalpy, cellMat);
530  }
531 
532  return true;
533  }

672  void Optimizer::fixTemplateLengths()
673  {
674  int steps = getNumberOfOptSteps();
675  if (steps < 1) {
676  steps = 1;
677  }
678 
679  // Optimizer:
680  QStringList filenames = m_templates.keys();
681  for (QStringList::const_iterator
682  it = filenames.constBegin(),
683  it_end = filenames.constEnd();
684  it != it_end;
685  ++it) {
686  QStringList &current = m_templates[*it];
687  while (current.size() > steps) {
688  current.removeLast();
689  }
690  while (current.size() < steps) {
691  current << "";
692  }
693  }
694 
695  // Optimizer:
696  filenames = m_QITemplates.keys();
697  for (QStringList::const_iterator
698  it = filenames.constBegin(),
699  it_end = filenames.constEnd();
700  it != it_end;
701  ++it) {
702  QStringList &current = m_QITemplates[*it];
703  while (current.size() > steps) {
704  current.removeLast();
705  }
706  while (current.size() < steps) {
707  current << "";
708  }
709  }
710  }
