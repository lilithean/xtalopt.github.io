28  // Since QProcess doesn't support polling at all (e.g. the only way
29  // to track its progress is to monitor signals), this class provides
30  // a basic polling wrapper.
31  class LocalQueueProcess : public QProcess
32  { Q_OBJECT;
33  public:
34  enum Status {NotStarted = 0, Running, Finished, Error};
35  LocalQueueProcess(QObject *parent) :
36  QProcess(parent), m_status(NotStarted)
37  {
38  connect(this, SIGNAL(started()), this, SLOT(setRunning()));
39  connect(this, SIGNAL(finished(int, QProcess::ExitStatus)),
40  this, SLOT(setFinished()));
41  connect(this, SIGNAL(error(QProcess::ProcessError)),
42  this, SLOT(setFinished()));}
43  public slots:
44  void setRunning() {m_status = Running;};
45  void setFinished() {m_status = Finished;};
46  Status status() {return m_status;};
47  private:
48  Status m_status;
49  };
