152  void AbstractEditTab::updateGUI()
153  {
154  if (!m_isInitialized) {
155  return;
156  }
157  Q_ASSERT_X(m_queueInterfaces.contains(m_opt->queueInterface()) ||
158  m_opt->queueInterface() == 0, Q_FUNC_INFO,
159  "Current queue interface is unknown to AbstractEditTab.");
160  Q_ASSERT_X(m_optimizers.contains(m_opt->optimizer()) ||
161  m_opt->optimizer() == 0, Q_FUNC_INFO,
162  "Current optimizer is unknown to AbstractEditTab.");
163  Q_ASSERT(m_optimizers.size() == ui_combo_optimizers->count());
164  Q_ASSERT(m_queueInterfaces.size() == ui_combo_queueInterfaces->count());
165 
166  if (m_opt->optimizer()) {
167  int optIndex = m_optimizers.indexOf(m_opt->optimizer());
168  ui_combo_optimizers->setCurrentIndex(optIndex);
169  }
170 
171  if (m_opt->queueInterface()) {
172  int qiIndex = m_queueInterfaces.indexOf(m_opt->queueInterface());
173  ui_combo_queueInterfaces->setCurrentIndex(qiIndex);
174  if (m_opt->queueInterface()->hasDialog()) {
175  ui_push_queueInterfaceConfig->setEnabled(true);
176  } else {
177  ui_push_queueInterfaceConfig->setEnabled(false);
178  }
179  }
180  else {
181  ui_push_queueInterfaceConfig->setEnabled(false);
182  }
183 
184  populateTemplates();
185  populateOptStepList();
186 
187  updateEditWidget();
188 
189  ui_edit_user1->setText( m_opt->optimizer()->getUser1());
190  ui_edit_user2->setText( m_opt->optimizer()->getUser2());
191  ui_edit_user3->setText( m_opt->optimizer()->getUser3());
192  ui_edit_user4->setText( m_opt->optimizer()->getUser4());
193  }

208  void AbstractEditTab::updateQueueInterface()
209  {
210  Q_ASSERT_X(m_queueInterfaces.contains(m_opt->queueInterface()) ||
211  m_opt->queueInterface() == 0, Q_FUNC_INFO,
212  "Current queue interface is unknown to AbstractEditTab.");
213 
214  unsigned int newQiIndex = ui_combo_queueInterfaces->currentIndex();
215 
216  Q_ASSERT(newQiIndex <= m_queueInterfaces.size() - 1);
217 
218  // Check that queueInterface has actually changed
219  if (m_queueInterfaces.indexOf(m_opt->queueInterface()) ==
220  newQiIndex &&
221  m_opt->queueInterface() != 0) {
222  return;
223  }
224 
225  QueueInterface *qi = m_queueInterfaces[newQiIndex];
226 
227  if (qi->hasDialog()) {
228  ui_push_queueInterfaceConfig->setEnabled(true);
229  } else {
230  ui_push_queueInterfaceConfig->setEnabled(false);
231  }
232 
233  emit queueInterfaceChanged(qi);
234  }

311  void AbstractEditTab::updateEditWidget()
312  {
313  if (!m_isInitialized) {
314  return;
315  }
316  QStringList filenames = getTemplateNames();
317  int templateInd = ui_combo_templates->currentIndex();
318  QString templateName = ui_combo_templates->currentText();
319  Q_ASSERT(templateInd >= 0 && templateInd < filenames.size());
320  Q_ASSERT(templateName.compare(filenames.at(templateInd)) == 0);
321 
322  if (m_opt->optimizer()->getNumberOfOptSteps() !=
323  ui_list_optStep->count()) {
324  populateOptStepList();
325  }
326 
327  int optStepIndex = ui_list_optStep->currentRow();
328  Q_ASSERT(optStepIndex >= 0 &&
329  optStepIndex < m_opt->optimizer()->getNumberOfOptSteps());
330 
331  // Display appropriate entry widget. Only text entry is supported
332  // by default, reimplement this function in the derived class if
333  // list entry is needed and just call
334  // AbstractEditTab::updateEditWidget to handle entries
335  // requiring text entry.
336  ui_list_edit->setVisible(false);
337  ui_edit_edit->setVisible(true);
338 
339  // Update text edit widget
340  Q_ASSERT(getTemplateNames().contains(templateName));
341  QString text = m_opt->optimizer()->getTemplate(templateName, optStepIndex);
342 
343  ui_edit_edit->blockSignals(true);
344  ui_edit_edit->setText(text);
345  ui_edit_edit->setCurrentFont(QFont("Courier"));
346  ui_edit_edit->blockSignals(false);
347  }

384  void AbstractEditTab::populateOptStepList()
385  {
386  if (!m_isInitialized) {
387  return;
388  }
389  ui_list_optStep->blockSignals(true);
390  ui_list_optStep->clear();
391 
392  // Just return if either QI or optimizer are missing
393  if (!m_opt->optimizer() || !m_opt->queueInterface()) {
394  ui_list_optStep->blockSignals(false);
395  return;
396  }
397 
398  int currentOptStep = ui_list_optStep->currentRow();
399  const int maxSteps = m_opt->optimizer()->getNumberOfOptSteps();
400  if (currentOptStep < 0) currentOptStep = 0;
401  if (currentOptStep >= maxSteps) currentOptStep = maxSteps - 1;
402 
403  for (int i = 1; i <= maxSteps; ++i) {
404  ui_list_optStep->addItem(tr("Optimization %1").arg(i));
405  }
406  ui_list_optStep->blockSignals(false);
407 
408  ui_list_optStep->setCurrentRow(currentOptStep);
409  }

411  void AbstractEditTab::appendOptStep()
412  {
413  // Reimplement in derived class if Optimizer generic data is needed
414  const int maxSteps = m_opt->optimizer()->getNumberOfOptSteps();
415  const int currentOptStep = ui_list_optStep->currentRow();
416  Q_ASSERT(currentOptStep >= 0 && currentOptStep < maxSteps);
417 
418  // Copy the current files into a new entry at the end of the opt step list
419  // Optimizer templates
420  QStringList templates = getTemplateNames();
421 
422  for (QStringList::const_iterator
423  it = templates.constBegin(),
424  it_end = templates.constEnd();
425  it != it_end;
426  ++it) {
427  m_opt->optimizer()->
428  appendTemplate(*it, m_opt->optimizer()->
429  getTemplate(*it, currentOptStep));
430  }
431 
432  populateOptStepList();
433  }
