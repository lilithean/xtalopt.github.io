205  void QueueManager::checkPopulation()
206  {
207  // Count jobs
208  uint running = 0;
209  uint optimized = 0;
210  uint submitted = 0;
211  m_tracker->lockForRead();
212  QList<Structure*> structures = *m_tracker->list();
213  m_tracker->unlock();
214 
215  // Check to see that the number of running jobs is >= that specified:
216  Structure *structure = 0;
217  Structure::State state;
218  int fail=0;
219  for (int i = 0; i < structures.size(); ++i) {
220  structure = structures.at(i);
221  structure->lock()->lockForRead();
222  state = structure->getStatus();
223  if (structure->getFailCount() != 0) fail++;
224  structure->lock()->unlock();
225  // Count submitted structures
226  if ( state == Structure::Submitted ||
227  state == Structure::InProcess ){
228  m_runningTracker.lockForWrite();
229  m_runningTracker.append(structure);
230  m_runningTracker.unlock();
231  submitted++;
232  }
233  // Count running jobs and update trackers
234  if ( state != Structure::Optimized &&
235  state != Structure::Duplicate &&
236  state != Structure::Killed &&
237  state != Structure::Removed ) {
238  running++;
239  m_runningTracker.lockForWrite();
240  m_runningTracker.append(structure);
241  m_runningTracker.unlock();
242  }
243  else {
244  if ( state == Structure::Optimized ) {
245  optimized++;
246  }
247  m_runningTracker.lockForWrite();
248  m_runningTracker.remove(structure);
249  m_runningTracker.unlock();
250  }
251  }
252  emit newStatusOverview(optimized, running, fail);
253 
254  // Submit any jobs if needed
255  m_jobStartTracker.lockForWrite();
256  int pending = m_jobStartTracker.list()->size();
257  if (pending != 0 &&
258  (
259  !m_opt->limitRunningJobs ||
260  submitted < m_opt->runningJobLimit
261  )) {
262  // Submit a single throttled job (1 submission per 3-8 seconds) if using
263  // a remote queue interface. Interval is randomly chosen each iteration.
264  // This prevents hammering the pbs server from multiple XtalOpt instances
265  // if there is a problem with the queue.
266  if (qobject_cast<RemoteQueueInterface*>
267  (m_opt->queueInterface()) != NULL) {
268  if (m_lastSubmissionTimeStamp->secsTo(QDateTime::currentDateTime())
269  >= 3 + (6 * RANDDOUBLE())) {
270  startJob();
271  ++submitted;
272  --pending;
273  *m_lastSubmissionTimeStamp = QDateTime::currentDateTime();
274  }
275  }
276  else {
277  // Local job submission doesn't need to be throttled
278  while (pending != 0 &&
279  (
280  !m_opt->limitRunningJobs ||
281  submitted < m_opt->runningJobLimit
282  )
283  ) {
284  startJob();
285  ++submitted;
286  --pending;
287  }
288  }
289  }
290  m_jobStartTracker.unlock();
291 
292  // Generate requests
293  m_tracker->lockForWrite(); // Write lock for m_requestedStructures var
294  m_newStructureTracker.lockForRead();
295 
296  // Avoid convience function calls here, as occaisional deadlocks
297  // can occur.
298  //
299  // total is getAllStructures().size() + m_requestedStructures;
300  int total = m_tracker->size() + m_newStructureTracker.size()
301  + m_requestedStructures;
302  // incomplete is getAllRunningStructures.size() + m_requestedStructures:
303  int incomplete = m_runningTracker.size() + m_newStructureTracker.size()
304  + m_requestedStructures;
305  int needed = m_opt->contStructs - incomplete;
306 
307  if (
308  // Are we at the continuous structure limit?
309  ( needed > 0) &&
310  // Is the cutoff either disabled or reached/exceeded?
311  ( m_opt->cutoff <= 0 || total < m_opt->cutoff) &&
312  // Check if we are testing. If so, have we reached the testing limit?
313  ( !m_opt->testingMode || total < m_opt->test_nStructs)
314  ) {
315  // emit requests
316  qDebug() << "Need " << needed << " structures. " << incomplete << " already incomplete.";
317  for (int i = 0; i < needed; ++i) {
318  ++m_requestedStructures;
319  emit needNewStructure();
320  qDebug() << "Requested new structure. Total requested: " << m_requestedStructures;
321  }
322  }
323 
324  m_newStructureTracker.unlock();
325  m_tracker->unlock();
326  return;
327  }

329  void QueueManager::checkRunning()
330  {
331  // Ensure that this is only called from the QM thread:
332  Q_ASSERT_X(QThread::currentThread() == m_thread, Q_FUNC_INFO,
333  "Attempting to run QueueManager::checkRunning "
334  "from a thread other than the QM thread. "
335  );
336 
337  // Get list of running structures
338  QList<Structure*> runningStructures = getAllRunningStructures();
339 
340  // iterate over all structures and handle each based on its status
341  for (QList<Structure*>::iterator
342  s_it = runningStructures.begin(),
343  s_it_end = runningStructures.end();
344  s_it != s_it_end;
345  ++s_it) {
346 
347  // Assign pointer for convenience
348  Structure *structure = *s_it;
349 
350  // Check if this structure has any handlers pending. Skip if so.
351  if (m_newlyOptimizedTracker.contains(structure) ||
352  m_stepOptimizedTracker.contains(structure) ||
353  m_inProcessTracker.contains(structure) ||
354  m_errorTracker.contains(structure) ||
355  m_submittedTracker.contains(structure) ||
356  m_newlyKilledTracker.contains(structure) ||
357  m_newDuplicateTracker.contains(structure) ||
358  m_restartTracker.contains(structure) ||
359  m_newSubmissionTracker.contains(structure)) {
360  continue;
361  }
362 
363  // Lookup status
364  structure->lock()->lockForRead();
365  Structure::State status = structure->getStatus();
366  structure->lock()->unlock();
367 
368  // Check status
369  switch (status) {
370  case Structure::InProcess:
371  handleInProcessStructure(structure);
372  break;
373  case Structure::WaitingForOptimization:
374  handleWaitingForOptimizationStructure(structure);
375  break;
376  case Structure::StepOptimized:
377  handleStepOptimizedStructure(structure);
378  break;
379  case Structure::Optimized:
380  // Shouldn't happen -- this is called by handleStepOptimizedStructure
381  // when needed. There is a race condition between the check* functions
382  // -- The structure may be removed from the list of running structures
383  // by checkPopulation before checkRunning is called.
384  //handleOptimizedStructure(structure);
385  break;
386  case Structure::Error:
387  handleErrorStructure(structure);
388  break;
389  case Structure::Submitted:
390  handleSubmittedStructure(structure);
391  break;
392  case Structure::Killed:
393  handleKilledStructure(structure);
394  break;
395  case Structure::Removed:
396  handleRemovedStructure(structure);
397  break;
398  case Structure::Restart:
399  handleRestartStructure(structure);
400  break;
401  case Structure::Updating:
402  handleUpdatingStructure(structure);
403  break;
404  case Structure::Duplicate:
405  handleDuplicateStructure(structure);
406  break;
407  case Structure::Empty:
408  handleEmptyStructure(structure);
409  break;
410  }
411  }
412 
413  return;
414  }

571  void QueueManager::handleErrorStructure_(Structure *s)
572  {
573  Q_ASSERT(trackerContainsStructure(s, &m_errorTracker));
574  removeFromTrackerWhenScopeEnds popper (s, &m_errorTracker);
575 
576  if (s->getStatus() != Structure::Error) {
577  return;
578  }
579 
580  stopJob(s);
581 
582  // Lock for writing
583  QWriteLocker locker (s->lock());
584 
585  s->addFailure();
586 
587  // If the number of failures has exceed the limit, take
588  // appropriate action
589  if (s->getFailCount() >= m_opt->failLimit) {
590  switch (OptBase::FailActions(m_opt->failAction)) {
591  case OptBase::FA_DoNothing:
592  default:
593  // resubmit job
594  s->setStatus(Structure::Restart);
595  emit structureUpdated(s);
596  return;
597  case OptBase::FA_KillIt:
598  locker.unlock();
599  killStructure(s);
600  emit structureUpdated(s);
601  return;
602  case OptBase::FA_Randomize:
603  s->setStatus(Structure::Empty);
604  locker.unlock();
605  m_opt->replaceWithRandom(s, tr("excessive failures"));
606  s->setStatus(Structure::Restart);
607  emit structureUpdated(s);
608  case OptBase::FA_NewOffspring:
609  s->setStatus(Structure::Empty);
610  locker.unlock();
611  m_opt->replaceWithOffspring(s, tr("excessive failures"));
612  s->setStatus(Structure::Restart);
613  emit structureUpdated(s);
614  return;
615  }
616  }
617  // Resubmit job if failure limit hasn't been reached
618  else {
619  s->setStatus(Structure::Restart);
620  emit structureUpdated(s);
621  return;
622  }
623  }
