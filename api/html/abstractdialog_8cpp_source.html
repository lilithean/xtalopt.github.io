175  void AbstractDialog::resumeSession()
176  {
177  QString filename;
178  QFileDialog dialog (NULL,
179  QString("Select .state file to resume"),
180  m_opt->filePath,
181  "*.state;;*.*");
182  dialog.selectFile(tr("%1.state")
183  .arg(m_opt->getIDString().toLower()));
184  dialog.setFileMode(QFileDialog::ExistingFile);
185  if (dialog.exec())
186  filename = dialog.selectedFiles().first();
187  else { // User cancel file selection.
188  return;
189  }
190 
191  QtConcurrent::run(this, &AbstractDialog::resumeSession_, filename);
192  }

194  void AbstractDialog::resumeSession_(const QString &filename)
195  {
196  m_opt->emitStartingSession();
197  startProgressUpdate(tr("Resuming session..."), 0, 0);
198  m_opt->tracker()->lockForWrite();
199  m_opt->tracker()->deleteAllStructures();
200  m_opt->tracker()->unlock();
201  if (!m_opt->load(filename)) {
202  stopProgressUpdate();
203  m_opt->isStarting = false;
204  return;
205  }
206  // Refresh dialog and settings
207  writeSettings();
208  stopProgressUpdate();
209 
210  // Emit session started signals
211  if (m_opt->readOnly)
212  m_opt->emitReadOnlySessionStarted();
213  else
214  m_opt->emitSessionStarted();
215  }
