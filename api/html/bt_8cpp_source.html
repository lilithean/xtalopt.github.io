45 QStringList getBackTrace()
46 {
47  //now get the backtrace of the code at this point
48  //(we can only do this if we have 'execinfo.h'
49 #ifdef _HAVE_EXECINFO_H_
50 
51  //create a void* array to hold the function addresses. We will only go at most 25 deep
52  const int maxdepth(25);
53 
54  void *func_addresses[maxdepth];
55  int nfuncs = backtrace(func_addresses, maxdepth);
56 
57  //now get the function names associated with these symbols. This should work for elf
58  //binaries, though additional linker options may need to have been called
59  //(e.g. -rdynamic for GNU ld. See the glibc documentation for 'backtrace')
60  char **symbols = backtrace_symbols(func_addresses, nfuncs);
61 
62  //save all of the function names onto the QStringList....
63  //(note that none of this will work if we have run out of memory)
64  QStringList ret;
65 
66  for (int i=0; i<nfuncs; i++)
67  {
68  if (btregexp.indexIn(symbols[i]) != -1)
69  {
70  //get the library or app that contains this symbol
71  QString unit = btregexp.cap(1);
72  //get the symbol
73  QString symbol = btregexp.cap(2);
74  //get the offset
75  QString offset = btregexp.cap(3);
76  //get the address
77  QString address = btregexp.cap(4);
78 
79  //now try and demangle the symbol
80  int stat;
81  char *demangled =
82  abi::__cxa_demangle(qPrintable(symbol),0,0,&stat);
83 
84  if (demangled)
85  {
86  symbol = demangled;
87  delete demangled;
88  }
89 
90  //put this all together
91  ret.append( QString("%1: %2 %3").arg(unit,symbol,address) );
92  }
93  else
94  {
95  //I don't recognise this string - just add the raw
96  //string to the backtrace
97  ret.append(symbols[i]);
98  }
99  }
100 
101  //we now need to release the memory of the symbols array. Since it was allocated using
102  //malloc, we must release it using 'free'
103  delete symbols;
104 
105  return ret;
106

109  "Backtrace is not available without execinfo.h")
