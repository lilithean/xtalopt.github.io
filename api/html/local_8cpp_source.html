66  bool LocalQueueInterface::isReadyToSearch(QString *str)
67  {
68  // Is a working directory specified?
69  if (m_opt->filePath.isEmpty()) {
70  *str = tr("Local working directory is not set. Check your Queue "
71  "configuration.");
72  return false;
73  }
74 
75  // Can we write to the working directory?
76  QDir workingdir (m_opt->filePath);
77  bool writable = true;
78  if (!workingdir.exists()) {
79  if (!workingdir.mkpath(m_opt->filePath)) {
80  writable = false;
81  }
82  }
83  else {
84  // If the path exists, attempt to open a small test file for writing
85  QString filename = m_opt->filePath + QString("queuetest-")
86  + QString::number(RANDUINT());
87  QFile file (filename);
88  if (!file.open(QFile::ReadWrite)) {
89  writable = false;
90  }
91  file.remove();
92  }
93  if (!writable) {
94  *str = tr("Cannot write to working directory '%1'.\n\nPlease "
95  "change the permissions on this directory or use "
96  "a different one.").arg(m_opt->filePath);
97  return false;
98  }
99 
100  *str = "";
101  return true;

104  bool LocalQueueInterface::writeFiles
105  (Structure *s, const QHash<QString, QString> &fileHash) const
106  {
107  // Create file objects
108  QList<QFile*> files;
109  QStringList filenames = fileHash.keys();
110  for (int i = 0; i < filenames.size(); i++) {
111  files.append(new QFile (s->fileName() + "/" + filenames.at(i)));
112  }
113 
114  // Check that the files can be written to
115  for (int i = 0; i < files.size(); i++) {
116  if (!files.at(i)->open( QIODevice::WriteOnly | QIODevice::Text ) ) {
117  m_opt->error(tr("Cannot write input file %1 (file writing failure)", "1 is a file path").arg(files.at(i)->fileName()));
118  qDeleteAll(files);
119  return false;
120  }
121  }
122 
123  // Set up text streams
124  QList<QTextStream*> streams;
125  for (int i = 0; i < files.size(); i++) {
126  streams.append(new QTextStream (files.at(i)));
127  }
128 
129  // Write files
130  for (int i = 0; i < streams.size(); i++) {
131  *(streams.at(i)) << fileHash[filenames.at(i)];
132  }
133 
134  // Close files
135  for (int i = 0; i < files.size(); i++) {
136  files.at(i)->close();
137  }
138 
139  // Clean up
140  qDeleteAll(streams);
141  qDeleteAll(files);
142  return true;
143  }

145  bool LocalQueueInterface::startJob(Structure *s)
146  {
147  if (s->getJobID() != 0) {
148  m_opt->warning(tr("LocalQueueInterface::startJob: Attempting to start "
149  "job for structure %1, but a JobID is already set (%2)."
150  ).arg(s->getIDString()).arg(s->getJobID()));
151  return false;
152  }
153  // TODO the corresponding function in Optimizer should prepend a
154  // path for e.g. windows
155  QString command = m_opt->optimizer()->localRunCommand();
156 
157 #ifdef WIN32
158  command = "cmd.exe /C \"" + command + "\"";
159 #endif // WIN32
160 
161  LocalQueueProcess *proc = new LocalQueueProcess(this);
162  proc->setWorkingDirectory(s->fileName());
163  if (!m_opt->optimizer()->stdinFilename().isEmpty()) {
164  proc->setStandardInputFile(s->fileName()
165  + "/" + m_opt->optimizer()->stdinFilename());
166  }
167  if (!m_opt->optimizer()->stdoutFilename().isEmpty()) {
168  proc->setStandardOutputFile(s->fileName()
169  + "/" + m_opt->optimizer()->stdoutFilename());
170  }
171  if (!m_opt->optimizer()->stderrFilename().isEmpty()) {
172  proc->setStandardErrorFile(s->fileName()
173  + "/" + m_opt->optimizer()->stderrFilename());
174  }
175 
176  proc->start(command);
177 
178 #ifdef WIN32
179  unsigned long pid = proc->pid()->dwProcessId;
180 #else // WIN32
181  unsigned long pid = proc->pid();
182 #endif // WIN32
183 
184  s->startOptTimer();
185  s->setJobID(pid);
186 
187  m_processes.insert(pid, proc);
188 
189  return true;
190  }

192  bool LocalQueueInterface::stopJob(Structure *s)
193  {
194  QWriteLocker wLocker (s->lock());
195 
196  unsigned long pid = static_cast<unsigned long>(s->getJobID());
197 
198  if (pid == 0) {
199  // The job is not running, so just return
200  return true;
201  }
202 
203  // Look-up process instance
204  LocalQueueProcess *proc = m_processes.value(pid, 0);
205  if (!proc) {
206  // No process is associated with this JobID.
207  return true;
208  }
209 
210  // Kill job
211  proc->kill();
212  s->setJobID(0);
213  s->stopOptTimer();
214  m_processes.remove(pid);
215  return true;
216  }

218  QueueInterface::QueueStatus
219  LocalQueueInterface::getStatus(Structure *s) const
220  {
221  // lock Structure
222  QReadLocker wlocker (s->lock());
223 
224  // Look-up process instance
225  unsigned long pid = static_cast<unsigned long>(s->getJobID());
226 
227  // If jobID = 0 and structure is not in "Submitted" state, return an error.
228  if (!pid && s->getStatus() != Structure::Submitted) {
229  return QueueInterface::Error;
230  }
231 
232  // Look-up process instance
233  LocalQueueProcess *proc = m_processes.value(pid, 0);
234 
235  // If structure is submitted, check if it's process is in the
236  // lookup table. If not, check if the completion file has been
237  // written.
238  //
239  // If the completion file exists, then the job finished before the
240  // queue checks could see it, and the function will continue on to
241  // the status checks below.
242  //
243  // Is the structure in Submitted state?
244  if (s->getStatus() == Structure::Submitted) {
245  // If the process isn't in the table
246  if (pid == 0 || proc == 0) {
247  // Is the output file exist absent?
248  bool exists;
249  m_opt->optimizer()->checkIfOutputFileExists(s, &exists);
250  if (!exists) {
251  // The output file does not exist -- the job is still
252  // pending.
253  return QueueInterface::Pending;
254  }
255  }
256  // The job is either running or finished
257  return QueueInterface::Started;
258  }
259 
260  // If the process is not in the table, return an error
261  if (!proc) {
262  return QueueInterface::Error;
263  }
264 
265  // Note that this is not part of QProcess - status() is defined in
266  // LocalQueueProcess
267  switch (proc->status()) {
268  case LocalQueueProcess::NotStarted:
269  return QueueInterface::Pending;
270  case LocalQueueProcess::Running:
271  return QueueInterface::Running;
272  case LocalQueueProcess::Finished:
273  // Was the run successful?
274  if (proc->exitCode() != 0) {
275  m_opt->warning(tr("%1: Structure %2, PID=%3 failed. QProcess error "
276  "code: %4. Process exit code: %5 errStr: %6\n"
277  "stdout:\n%7\nstderr:\n%8")
278  .arg(Q_FUNC_INFO).arg(s->getIDString()).arg(pid)
279  .arg(proc->error()).arg(proc->exitCode())
280  .arg(proc->errorString())
281  .arg(QString(proc->readAllStandardOutput()))
282  .arg(QString(proc->readAllStandardError())));
283  return QueueInterface::Error;
284  }
285  bool success;
286  m_opt->optimizer()->checkForSuccessfulOutput(s, &success);
287  if (success) {
288  return QueueInterface::Success;
289  }
290  else {
291  return QueueInterface::Error;
292  }
293  default:
294  // Shouldn't reach this point...
295  return QueueInterface::Unknown;
296  }
297  }

329  bool LocalQueueInterface::grepFile(Structure *s,
330  const QString &matchText,
331  const QString &filename,
332  QStringList *matches,
333  int *exitcode,
334  const bool caseSensitive) const
335  {
336  if (exitcode) {
337  *exitcode = 1;
338  }
339  QStringList list;
340  QString contents;
341  if (!fetchFile(s, filename, &contents)) {
342  if (exitcode) {
343  *exitcode = 2;
344  }
345  return false;
346  }
347 
348  list = contents.split("\n", QString::SkipEmptyParts);
349 
350  Qt::CaseSensitivity cs;
351  if (caseSensitive) {
352  cs = Qt::CaseSensitive;
353  }
354  else {
355  cs = Qt::CaseInsensitive;
356  }
357 
358  for (QStringList::const_iterator
359  it = list.begin(),
360  it_end = list.end();
361  it != it_end;
362  ++it) {
363  if ((*it).contains(matchText, cs)) {
364  if (exitcode) {
365  *exitcode = 0;
366  }
367  if (matches) {
368  *matches << *it;
369  }
370  }
371  }
372 
373  return true;
374  }
