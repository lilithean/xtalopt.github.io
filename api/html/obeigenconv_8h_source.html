30  inline Eigen::Matrix3d OB2Eigen(const OpenBabel::matrix3x3 &obm)
31  {
32  Eigen::Matrix3d mat;
33  mat(0,0)=obm.Get(0,0);mat(0,1)=obm.Get(0,1);mat(0,2)=obm.Get(0,2);
34  mat(1,0)=obm.Get(1,0);mat(1,1)=obm.Get(1,1);mat(1,2)=obm.Get(1,2);
35  mat(2,0)=obm.Get(2,0);mat(2,1)=obm.Get(2,1);mat(2,2)=obm.Get(2,2);

39  inline OpenBabel::matrix3x3 Eigen2OB(const Eigen::Matrix3d &mat)
40  {
41  OpenBabel::matrix3x3 obm;
42  obm.Set(0,0,mat(0,0));obm.Set(0,1,mat(0,1));obm.Set(0,2,mat(0,2));
43  obm.Set(1,0,mat(1,0));obm.Set(1,1,mat(1,1));obm.Set(1,2,mat(1,2));
44  obm.Set(2,0,mat(2,0));obm.Set(2,1,mat(2,1));obm.Set(2,2,mat(2,2));
45  return obm;
46  }
