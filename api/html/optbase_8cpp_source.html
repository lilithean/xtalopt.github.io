149  QList<double> OptBase::getProbabilityList(const QList<Structure*> &structures) {
150  // IMPORTANT: structures must contain one more structure than
151  // needed -- the last structure in the list will be removed from
152  // the probability list!
153  if (structures.size() <= 2) {
154  return QList<double>();
155  }
156 
157  QList<double> probs;
158  Structure *s=0, *first=0, *last=0;
159  first = structures.first();
160  last = structures.last();
161  first->lock()->lockForRead();
162  last->lock()->lockForRead();
163  double lowest = first->getEnthalpy();
164  double highest = last->getEnthalpy();;
165  double spread = highest - lowest;
166  last->lock()->unlock();
167  first->lock()->unlock();
168  // If all structures are at the same enthalpy, lets save some time...
169  if (spread <= 1e-5) {
170  double dprob = 1.0/static_cast<double>(structures.size()-1);
171  double prob = 0;
172  for (int i = 0; i < structures.size()-1; i++) {
173  probs.append(prob);
174  prob += dprob;
175  }
176  return probs;
177  }
178  // Generate a list of floats from 0->1 proportional to the enthalpies;
179  // E.g. if enthalpies are:
180  // -5 -2 -1 3 5
181  // We'll have:
182  // 0 0.3 0.4 0.8 1
183  for (int i = 0; i < structures.size(); i++) {
184  s = structures.at(i);
185  s->lock()->lockForRead();
186  probs.append( ( s->getEnthalpy() - lowest ) / spread);
187  s->lock()->unlock();
188  }
189  // Subtract each value from one, and find the sum of the resulting list
190  // We'll end up with:
191  // 1 0.7 0.6 0.2 0 -- sum = 2.5
192  double sum = 0;
193  for (int i = 0; i < probs.size(); i++){
194  probs[i] = 1.0 - probs.at(i);
195  sum += probs.at(i);
196  }
197  // Normalize with the sum so that the list adds to 1
198  // 0.4 0.28 0.24 0.08 0
199  for (int i = 0; i < probs.size(); i++){
200  probs[i] /= sum;
201  }
202  // Then replace each entry with a cumulative total:
203  // 0.4 0.68 0.92 1 1
204  sum = 0;
205  for (int i = 0; i < probs.size(); i++){
206  sum += probs.at(i);
207  probs[i] = sum;
208  }
209  // Pop off the last entry (remember the n_popSize + 1 earlier?)
210  // 0.4 0.68 0.92 1
211  probs.removeLast();
212  // And we have a enthalpy weighted probability list! To use:
213  //
214  // double r = rand.NextFloat();
215  // uint ind;
216  // for (ind = 0; ind < probs.size(); ind++)
217  // if (r < probs.at(ind)) break;
218  //
219  // ind will hold the chosen index.
220  return probs;
221  }

223  bool OptBase::save(const QString &stateFilename, bool notify)
224  {
225  if (isStarting ||
226  readOnly) {
227  savePending = false;
228  return false;
229  }
230  QReadLocker trackerLocker (m_tracker->rwLock());
231  QMutexLocker locker (stateFileMutex);
232  QString filename;
233  if (stateFilename.isEmpty()) {
234  filename = filePath + "/" + m_idString.toLower() + ".state";
235  }
236  else {
237  filename = stateFilename;
238  }
239  QString oldfilename = filename + ".old";
240 
241  if (notify) {
242  m_dialog->startProgressUpdate(tr("Saving: Writing %1...")
243  .arg(filename),
244  0, 0);
245  }
246 
247  // Copy .state -> .state.old
248  if (QFile::exists(filename) ) {
249  if (QFile::exists(oldfilename)) {
250  QFile::remove(oldfilename);
251  }
252  QFile::copy(filename, oldfilename);
253  }
254 
255  SETTINGS(filename);
256  const int VERSION = m_schemaVersion;
257  settings->beginGroup(m_idString.toLower());
258  settings->setValue("version", VERSION);
259  settings->setValue("saveSuccessful", false);
260  settings->endGroup();
261 
262  // Write/update .state
263  m_dialog->writeSettings(filename);
264 
265  // Loop over structures and save them
266  QList<Structure*> *structures = m_tracker->list();
267 
268  QString structureStateFileName;
269 
270  Structure* structure;
271  for (int i = 0; i < structures->size(); i++) {
272  structure = structures->at(i);
273  structure->lock()->lockForRead();
274  // Set index here -- this is the only time these are written, so
275  // this is "ok" under a read lock because of the savePending logic
276  structure->setIndex(i);
277  structureStateFileName = structure->fileName() + "/structure.state";
278  if (notify) {
279  m_dialog->updateProgressLabel(tr("Saving: Writing %1...")
280  .arg(structureStateFileName));
281  }
282  structure->writeSettings(structureStateFileName);
283  structure->lock()->unlock();
284  }
285 
287  // Print results files //
289 
290  QFile file (filePath + "/results.txt");
291  QFile oldfile (filePath + "/results_old.txt");
292  if (notify) {
293  m_dialog->updateProgressLabel(tr("Saving: Writing %1...")
294  .arg(file.fileName()));
295  }
296  if (oldfile.open(QIODevice::ReadOnly))
297  oldfile.remove();
298  if (file.open(QIODevice::ReadOnly))
299  file.copy(oldfile.fileName());
300  file.close();
301  if (!file.open(QIODevice::WriteOnly)) {
302  error("OptBase::save(): Error opening file "+file.fileName()+" for writing...");
303  savePending = false;
304  return false;
305  }
306  QTextStream out (&file);
307 
308  QList<Structure*> sortedStructures;
309 
310  for (int i = 0; i < structures->size(); i++)
311  sortedStructures.append(structures->at(i));
312  if (sortedStructures.size() != 0) {
313  Structure::sortAndRankByEnthalpy(&sortedStructures);
314  out << sortedStructures.first()->getResultsHeader() << endl;
315  }
316 
317  for (int i = 0; i < sortedStructures.size(); i++) {
318  structure = sortedStructures.at(i);
319  if (!structure) continue; // In case there was a problem copying.
320  structure->lock()->lockForRead();
321  out << structure->getResultsEntry() << endl;
322  structure->lock()->unlock();
323  if (notify) {
324  m_dialog->stopProgressUpdate();
325  }
326  }
327 
328  // Mark operation successful
329  settings->setValue(m_idString.toLower() + "/saveSuccessful", true);
330  DESTROY_SETTINGS(filename);
331 
332  savePending = false;
333  return true;
334  }

355  void OptBase::interpretKeyword_base(QString &line, Structure* structure)
356  {
357  QString rep = "";
358  // User data
359  if (line == "user1") rep += optimizer()->getUser1();
360  else if (line == "user2") rep += optimizer()->getUser2();
361  else if (line == "user3") rep += optimizer()->getUser3();
362  else if (line == "user4") rep += optimizer()->getUser4();
363  else if (line == "description") rep += description;
364  else if (line == "percent") rep += "%";
365 
366  // Structure specific data
367  if (line == "coords") {
368  QList<Avogadro::Atom*> atoms = structure->atoms();
369  QList<Avogadro::Atom*>::const_iterator it;
370  const Eigen::Vector3d *vec;
371  for (it = atoms.begin();
372  it != atoms.end();
373  it++) {
374  rep += static_cast<QString>(OpenBabel::etab.GetSymbol((*it)->atomicNumber())) + " ";
375  vec = (*it)->pos();
376  rep += QString::number(vec->x()) + " ";
377  rep += QString::number(vec->y()) + " ";
378  rep += QString::number(vec->z()) + "\n";
379  }
380  }
381  else if (line == "coordsInternalFlags") {
382  QList<Avogadro::Atom*> atoms = structure->atoms();
383  QList<Avogadro::Atom*>::const_iterator it;
384  const Eigen::Vector3d *vec;
385  for (it = atoms.begin();
386  it != atoms.end();
387  it++) {
388  rep += static_cast<QString>(OpenBabel::etab.GetSymbol((*it)->atomicNumber())) + " ";
389  vec = (*it)->pos();
390  rep += QString::number(vec->x()) + " 1 ";
391  rep += QString::number(vec->y()) + " 1 ";
392  rep += QString::number(vec->z()) + " 1\n";
393  }
394  }
395  else if (line == "coordsSuffixFlags") {
396  QList<Avogadro::Atom*> atoms = structure->atoms();
397  QList<Avogadro::Atom*>::const_iterator it;
398  const Eigen::Vector3d *vec;

405  rep += QString::number(vec->y()) + " ";
406  rep += QString::number(vec->z()) + " 1 1 1\n";
407  }
408  }
409  else if (line == "coordsId") {
410  QList<Avogadro::Atom*> atoms = structure->atoms();
411  QList<Avogadro::Atom*>::const_iterator it;
412  const Eigen::Vector3d *vec;

425  else if (line == "numSpecies") rep += QString::number(structure->getSymbols().size());
426  else if (line == "filename") rep += structure->fileName();
427  else if (line == "rempath") rep += structure->getRempath();
428  else if (line == "gen") rep += QString::number(structure->getGeneration());
429  else if (line == "id") rep += QString::number(structure->getIDNumber());
430  else if (line == "incar") rep += QString::number(structure->getCurrentOptStep());
431  else if (line == "optStep") rep += QString::number(structure->getCurrentOptStep());

433  if (!rep.isEmpty()) {
434  // Remove any trailing newlines

525  bool OptBase::createSSHConnections_libssh()
526  {
527  delete m_ssh;
528  SSHManagerLibSSH *libsshManager = new SSHManagerLibSSH(5, this);
529  m_ssh = libsshManager;
530  QString pw = "";
531  for (;;) {
532  try {
533  libsshManager->makeConnections(host, username, pw, port);
534  }
535  catch (SSHConnection::SSHConnectionException e) {
536  QString err;
537  switch (e) {
538  case SSHConnection::SSH_CONNECTION_ERROR:
539  case SSHConnection::SSH_UNKNOWN_ERROR:
540  default:
541  err = "There was a problem connection to the ssh server at "
542  + username + "@" + host + ":" + QString::number(port) + ". "
543  "Please check that all provided information is correct, and "
544  "attempt to log in outside of Avogadro before trying again.";
545  error(err);
546  return false;
547  case SSHConnection::SSH_UNKNOWN_HOST_ERROR: {
548  // The host is not known, or has changed its key.
549  // Ask user if this is ok.
550  err = "The host "
551  + host + ":" + QString::number(port)
552  + " either has an unknown key, or has changed it's key:\n"
553  + libsshManager->getServerKeyHash() + "\n"
554  + "Would you like to trust the specified host?";
555  bool ok;
556  // This is a BlockingQueuedConnection, which blocks until
557  // the slot returns.
558  emit needBoolean(err, &ok);
559  if (!ok) { // user cancels
560  return false;
561  }
562  libsshManager->validateServerKey();
563  continue;
564  } // end case
565  case SSHConnection::SSH_BAD_PASSWORD_ERROR: {
566  // Chances are that the pubkey auth was attempted but failed,
567  // so just prompt user for password.
568  err = "Please enter a password for "
569  + username + "@" + host + ":" + QString::number(port)
570  + ":";
571  bool ok;
572  QString newPassword;
573  // This is a BlockingQueuedConnection, which blocks until
574  // the slot returns.
575  emit needPassword(err, &newPassword, &ok);
576  if (!ok) { // user cancels
577  return false;
578  }
579  pw = newPassword;
580  continue;
581  } // end case
582  } // end switch
583  } // end catch
584  break;
585  } // end forever
586  return true;
587  }
