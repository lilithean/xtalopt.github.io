326  void Structure::readStructureSettings(const QString &filename)
327  {
328  SETTINGS(filename);
329  settings->beginGroup("structure");
330  int loadedVersion = settings->value("version", 0).toInt();
331  if (loadedVersion >= 1) { // Version 0 uses save(QTextStream)
332  setGeneration( settings->value("generation", 0).toInt());
333  setIDNumber( settings->value("id", 0).toInt());
334  setIndex( settings->value("index", 0).toInt());
335  setRank( settings->value("rank", 0).toInt());
336  setJobID( settings->value("jobID", 0).toInt());
337  setCurrentOptStep( settings->value("currentOptStep", 0).toInt());
338  setFailCount( settings->value("failCount", 0).toInt());
339  setParents( settings->value("parents", "").toString());
340  setRempath( settings->value("rempath", "").toString());
341  setStatus( State(settings->value("status", -1).toInt()));
342 
343  setOptTimerStart( QDateTime::fromString(settings->value("startTime", "").toString()));
344  setOptTimerEnd( QDateTime::fromString(settings->value("endTime", "").toString()));
345 
346  // History
347  settings->beginGroup("history");
348  // Atomic nums
349  int size, size2;
350  size = settings->beginReadArray("atomicNums");
351  m_histAtomicNums.clear();
352  for (int i = 0; i < size; i++) {
353  settings->setArrayIndex(i);
354  size2 = settings->beginReadArray(QString("atomicNums-%1").arg(i));
355  QList<unsigned int> cur;
356  for (int j = 0; j < size2; j++) {
357  settings->setArrayIndex(j);
358  cur.append(settings->value("value").toUInt());
359  }
360  settings->endArray();
361  m_histAtomicNums.append(cur);
362  }
363  settings->endArray();
364 
365  // Coords
366  size = settings->beginReadArray("coords");
367  m_histCoords.clear();
368  for (int i = 0; i < size; i++) {
369  settings->setArrayIndex(i);
370  size2 = settings->beginReadArray(QString("coords-%1").arg(i));
371  QList<Eigen::Vector3d> cur;
372  for (int j = 0; j < size2; j++) {
373  settings->setArrayIndex(j);
374  double x = settings->value("x").toDouble();
375  double y = settings->value("y").toDouble();
376  double z = settings->value("z").toDouble();
377  cur.append(Eigen::Vector3d(x, y, z));
378  }
379  settings->endArray();
380  m_histCoords.append(cur);
381  }
382  settings->endArray();
383 
384  // Energies
385  size = settings->beginReadArray("energies");
386  m_histEnergies.clear();
387  for (int i = 0; i < size; i++) {
388  settings->setArrayIndex(i);
389  m_histEnergies.append(settings->value("value").toDouble());
390  }
391  settings->endArray();
392 
393  // Enthalpies
394  size = settings->beginReadArray("enthalpies");
395  m_histEnthalpies.clear();
396  for (int i = 0; i < size; i++) {
397  settings->setArrayIndex(i);
398  m_histEnthalpies.append(settings->value("value").toDouble());
399  }
400  settings->endArray();
401 
402  // Cells
403  size = settings->beginReadArray("cells");
404  m_histCells.clear();
405  for (int i = 0; i < size; i++) {
406  settings->setArrayIndex(i);
407  Eigen::Matrix3d cur;
408  cur(0, 0) = settings->value("00").toDouble();
409  cur(0, 1) = settings->value("01").toDouble();
410  cur(0, 2) = settings->value("02").toDouble();
411  cur(1, 0) = settings->value("10").toDouble();
412  cur(1, 1) = settings->value("11").toDouble();
413  cur(1, 2) = settings->value("12").toDouble();
414  cur(2, 0) = settings->value("20").toDouble();
415  cur(2, 1) = settings->value("21").toDouble();
416  cur(2, 2) = settings->value("22").toDouble();
417  m_histCells.append(cur);
418  }
419  settings->endArray();
420 
421  settings->endGroup(); // history
422 
423  }
424  settings->endGroup();
425 
426  // Update config data
427  switch (loadedVersion) {
428  case 0: {
429  // Call load(QTextStream) to update
430  qDebug() << "Updating "
431  << filename
432  << " from Version 0 -> 1";
433  QFile file (filename);
434  file.open(QIODevice::ReadOnly);
435  QTextStream stream (&file);
436  load(stream);
437  }
438  case 1: // Histories added. Nothing to do, structures loaded will
439  // have empty histories
440  case 2:
441  default:
442  break;
443  }
444  }

451  void Structure::updateAndSkipHistory(const QList<unsigned int> &atomicNums,
452  const QList<Eigen::Vector3d> &coords,
453  const double energy,
454  const double enthalpy,
455  const Eigen::Matrix3d &cell)
456  {
457  Q_ASSERT_X(atomicNums.size() == coords.size() && coords.size() == numAtoms(),
458  Q_FUNC_INFO,
459  "Lengths of atomicNums and coords must match numAtoms().");
460 
461  // Update atoms
462  Atom *atom;
463  for (int i = 0; i < numAtoms(); i++) {
464  atom = atoms().at(i);
465  atom->setAtomicNumber(atomicNums.at(i));
466  atom->setPos(coords.at(i));
467  }
468 
469  // Update energy/enthalpy
470  if (fabs(enthalpy) < 1e-6) {
471  m_hasEnthalpy = false;
472  m_PV = 0.0;
473  }
474  else {
475  m_hasEnthalpy = true;
476  m_PV = enthalpy - energy;
477  }
478  m_enthalpy = enthalpy;
479  setEnergy(energy * EV_TO_KJ_PER_MOL);
480 
481  // Update cell if necessary
482  if (!cell.isZero()) {
483  OpenBabel::matrix3x3 obmat;
484  for (int i = 0; i < 3; i++) {
485  for (int j = 0; j < 3; j++) {
486  obmat.Set(i,j,cell(i,j));
487  }
488  }
489  OpenBabel::OBUnitCell *obcell = OBUnitCell();
490  obcell->SetData(obmat);
491  setOBUnitCell(obcell);
492  }
493 
494  emit moleculeChanged();
495  update();
496  }

591  bool Structure::addAtomRandomly(uint atomicNumber, double minIAD, double maxIAD, int maxAttempts, Atom **atom)
592  {
593  INIT_RANDOM_GENERATOR();
594  double IAD = -1;
595  int i = 0;
596  vector3 coords;
597 
598  // For first atom, add to 0, 0, 0
599  if (numAtoms() == 0) {
600  coords = vector3 (0,0,0);
601  }
602  else {
603  do {
604  // Generate random coordinates
605  IAD = -1;
606  double x = RANDDOUBLE() * radius() + maxIAD;
607  double y = RANDDOUBLE() * radius() + maxIAD;
608  double z = RANDDOUBLE() * radius() + maxIAD;
609 
610  coords.Set(x,y,z);
611  if (minIAD != -1) {
612  getNearestNeighborDistance(x, y, z, IAD);
613  }
614  else { break;};
615  i++;
616  } while (i < maxAttempts && IAD <= minIAD);
617 
618  if (i >= maxAttempts) return false;
619  }
620 
621  Atom *atm = addAtom();
622  atom = &atm;
623  Eigen::Vector3d pos (coords[0],coords[1],coords[2]);
624  (*atom)->setPos(pos);
625  (*atom)->setAtomicNumber(static_cast<int>(atomicNumber));
626  return true;
627  }

642  QString Structure::getResultsEntry() const
643  {

648  break;
649  case Killed:

652  break;
653  case Duplicate:
654  status = "Duplicate";
655  break;
656  case Error:
657  status = "Error";
658  break;

661  case InProcess:
662  case Empty:
663  case Updating:
664  case Submitted:
665  default:

667  break;

669  return QString("%1 %2 %3 %4 %5")
670  .arg(getRank(), 6)
671  .arg(getGeneration(), 6)
672  .arg(getIDNumber(), 6)
673  .arg(getEnthalpy(), 10)
674  .arg(status, 11);

897  bool Structure::generateIADHistogram(QList<QVariant> * distance,
898  QList<QVariant> * frequency,
899  double min, double max, double step,
900  Avogadro::Atom *atom) const
901  {
902  distance->clear();
903  frequency->clear();
904 
905  if (min > max && step > 0) {
906  qWarning() << "Structure::getNearestNeighborHistogram: min cannot be greater than max!";
907  return false;
908  }
909  if (step <= 0) {
910  qWarning() << "Structure::getNearestNeighborHistogram: invalid step size:" << step;
911  return false;
912  }
913 
914  double halfstep = step / 2.0;
915 
916  double val = min;
917  do {
918  distance->append(val);
919  frequency->append(0);
920  val += step;
921  } while (val < max);
922 
923  QList<Atom*> atomList = atoms();
924  QList<Vector3d> atomPositions;
925  for (int i = 0; i < atomList.size(); i++)
926  atomPositions.push_back(*(atomList.at(i)->pos()));
927 
928  Vector3d v1= atomPositions.at(0);
929  Vector3d v2= atomPositions.at(1);
930  double diff;
931 
932  // build histogram
933  // Loop over all atoms -- use map-reduce
934  if (atom == 0) {
935  QList<NNHistMap> ml;
936  for (int i = 0; i < atomList.size(); i++) {
937  NNHistMap m;
938  m.i = i; m.halfstep = halfstep; m.atomPositions = &atomPositions; m.dist = distance;
939  ml.append(m);
940  }
941  (*frequency) = QtConcurrent::blockingMappedReduced(ml, calcNNHistChunk, reduceNNHistChunks);
942  }
943  // Or, just the one requested
944  else {
945  v1 = *atom->pos();
946  for (int j = 0; j < atomList.size(); j++) {
947  if (atomList.at(j) == atom) continue;
948  v2 = atomPositions.at(j);
949  // Intercell
950  diff = fabs((v1-v2).norm());
951  for (int k = 0; k < distance->size(); k++) {
952  double radius = distance->at(k).toDouble();
953  double d;
954  if (diff != 0 && fabs(diff-radius) < halfstep) {
955  d = frequency->at(k).toDouble();
956  d++;
957  frequency->replace(k, d);
958  }
959  }
960  }
961  }
962 
963  return true;
964  }

966  bool Structure::compareIADDistributions(const std::vector<double> &d,
967  const std::vector<double> &f1,
968  const std::vector<double> &f2,
969  double decay,
970  double smear,
971  double *error)
972  {
973  // Check that smearing is possible
974  if (smear != 0 && d.size() <= 1) {
975  qWarning() << "Cluster::compareNNDist: Cannot smear with 1 or fewer points.";
976  return false;
977  }
978  // Check sizes
979  if (d.size() != f1.size() || f1.size() != f2.size()) {
980  qWarning() << "Cluster::compareNNDist: Vectors are not the same size.";
981  return false;
982  }
983 
984  // Perform a boxcar smoothing over range set by "smear"
985  // First determine step size of d, then convert smear to index units
986  double stepSize = fabs(d.at(1) - d.at(0));
987  int boxSize = ceil(smear/stepSize);
988  if (boxSize > d.size()) {
989  qWarning() << "Cluster::compareNNDist: Smear length is greater then d vector range.";
990  return false;
991  }
992  // Smear
993  vector<double> f1s, f2s, ds; // smeared vectors
994  if (smear != 0) {
995  double f1t, f2t, dt; // temporary variables
996  for (int i = 0; i < d.size() - boxSize; i++) {
997  f1t = f2t = dt = 0;
998  for (int j = 0; j < boxSize; j++) {
999  f1t += f1.at(i+j);
1000  f2t += f2.at(i+j);
1001  }
1002  f1s.push_back(f1t / double(boxSize));
1003  f2s.push_back(f2t / double(boxSize));
1004  ds.push_back(dt / double(boxSize));
1005  }
1006  } else {
1007  for (int i = 0; i < d.size() - boxSize; i++) {
1008  f1s.push_back(f1.at(i));
1009  f2s.push_back(f2.at(i));
1010  ds.push_back(d.at(i));
1011  }
1012  }
1013 
1014  // Calculate diff vector
1015  vector<double> diff;
1016  for (int i = 0; i < ds.size(); i++) {
1017  diff.push_back(fabs(f1s.at(i) - f2s.at(i)));
1018  }
1019 
1020  // Calculate decay function: Standard exponential decay with a
1021  // halflife of decay. If decay==0, no decay.
1022  double decayFactor = 0;
1023  // ln(2) / decay:
1024  if (decay != 0) {
1025  decayFactor = 0.69314718055994530941723 / decay;
1026  }
1027 
1028  // Calculate error:
1029  (*error) = 0;
1030  for (int i = 0; i < ds.size(); i++) {
1031  (*error) += exp(-decayFactor * ds.at(i)) * diff.at(i);
1032  }
1033 
1034  return true;
1035  }
